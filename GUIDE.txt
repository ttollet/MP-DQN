(For use in Ubuntu)
Add following to .bashrc, replacing [USER_HOME_DIR] in the final line with your respective home directory:

# pyenv
export PYENV_ROOT="$HOME/.pyenv"
command -v pyenv >/dev/null || export PATH="$PYENV_ROOT/bin:$PATH"
eval "$(pyenv init -)"

# pyenv-virtualenv
eval "$(pyenv virtualenv-init -)"

# poetry
export PATH=$PATH:/home/[USER_HOME_DIR]/.local/bin

-----------------------------
THEN

INSTALL PYENV & POETRY
curl https://pyenv.run | bash
pip3 install poetry

ENABLE THEM
source ~/.bashrc

INSTALL SOFTWARE
git clone https://github.com/ttollet/MP-DQN.git
cd MP-DQN/
poetry install
(Create a classic key with any password when prompted - it won't be needed)
poetry shell
./install_domains.sh

RUN WHAT WORKS - Currently 'goal' and 'platform' but not HFO ('soccer')
Visual results from P-DQN scripts:
python run_goal_pdqn.py
python run_platform_pdqn.py
